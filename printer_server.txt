sudo nano /etc/systemd/system/ticketprinter.service

-------------------------

[Unit]
Description=Flask Printer API
After=network-online.target
Wants=network-online.target

[Service]
User=restaurant-cloud
WorkingDirectory=/home/restaurant-cloud/ticket_printer/
ExecStart=/home/restaurant-cloud/posenv/bin/gunicorn -w 4 -b 0.0.0.0:5000 app:app
Environment="PYTHONUNBUFFERED=1"
Restart=always
RestartSec=5
TimeoutStartSec=30
StandardOutput=journal
StandardError=journal

[Install]
WantedBy=multi-user.target

-------------------------

sudo systemctl daemon-reload
sudo systemctl enable ticketprinter.service
sudo systemctl start ticketprinter.service
sudo systemctl status ticketprinter.service
